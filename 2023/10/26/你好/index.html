    /// <summary>
    /// 获取病人基本信息
    /// </summary>
    /// <param name="CardType"></param>
    /// <param name="CardNo"></param>
    /// <returns></returns>
    [HttpGet]
    public async Task<AjaxResult> GetHisPatientInfoByType(int CardType, string CardNo)
    {
        if (string.IsNullOrEmpty(CardNo))
        {
            return new AjaxResult("CardNo不能为空", AjaxResultType.Error);
        }
        if (CardType == 0)
        {
            return new AjaxResult("CardType传参错误", AjaxResultType.Error);
        }
        var result = await _patientService.GetHisPatientInfoByType(CardType, CardNo);
        return result.ToAjaxResult(true);
    }
